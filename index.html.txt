<script>
const moedas = [
    {
        numero: 1,
        nome: "Moeda Exemplo",
        descricao: "Descrição simples da moeda",
        detalhes: "Informações detalhadas da moeda."
    }
];

function carregarMoedas(lista = moedas) {
    const container = document.getElementById("listaMoedas");
    container.innerHTML = "";
    lista.forEach(moeda => {
        container.innerHTML += `
        <div class="card" onclick="abrirDetalhes(${moeda.numero})">
            <div class="coin-number">${moeda.numero}</div>
            <div class="image-box">Imagem da moeda ou da cédula</div>
            <strong>${moeda.nome}</strong>
        </div>`;
    });
}

function buscarMoeda(texto) {
    const filtradas = moedas.filter(m =>
        m.nome.toLowerCase().includes(texto.toLowerCase()) ||
        m.numero.toString().includes(texto)
    );
    carregarMoedas(filtradas);
}

function abrirDetalhes(numero) {
    const moeda = moedas.find(m => m.numero === numero);
    document.getElementById("detalhesMoeda").innerHTML = `
        <h3>${moeda.nome}</h3>
        <p>${moeda.detalhes}</p>
        <button onclick="fecharModal('modalDetalhes')">Fechar</button>
    `;
    document.getElementById("modalDetalhes").style.display = "flex";
}

function abrirSenha() {
    document.getElementById("modalSenha").style.display = "flex";
}

function verificarSenha() {
    const senha = document.getElementById("senha").value;
    if (senha === "murilolindo") {
        alert("Acesso liberado para registrar nova moeda.");
    } else {
        location.reload();
    }
}

function abrirTroca() {
    alert("Área de troca e venda.");
}

function abrirCuriosidades() {
    alert("Curiosidades da coleção.");
}

function fecharModal(id) {
    document.getElementById(id).style.display = "none";
}

carregarMoedas();
</script>
